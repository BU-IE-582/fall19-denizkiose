#Reading the data from csv files
myMatches <- read.csv(file="c:/Users/deniz/Desktop/matches.csv", header=TRUE, sep=",")

#We would like to predict whether more than 2.5 goals are going to be scored in a match (classification) and the number of goals (regression). For this reason we will use the statistics from the previous matches in myMatches.
#The response variable for the regression problem is the total number of goals (away and home) 
myData = myMatches
myData$resReg = myMatches$match_hometeam_score + myMatches$match_awayteam_score
myData = subset(myData, !is.na(resReg)) #omit matches with no final # of goals


#The response variable for the classification problem will be whether resreg (total number of goals) is more than 2.5 or not
#discretization of tıtal 3 of goals

for( i in 1:NROW(myData)){
  if(myData$resReg[i] > 2.5){ myData$resClas[i] <- 1}
  else { myData$resClas[i] <- 0}
}

avgGoals = mean(myData$resReg) #average number of goals
uniqueIdsHomeScore = aggregate(match_hometeam_score~match_hometeam_id, data=myData, FUN=function(x) c(mean=mean(x))) #average number of goals for each team when play at home
uniqueIdsAwayScore = aggregate(match_awayteam_score~match_awayteam_id, data=myData, FUN=function(x) c(mean=mean(x))) #average number of goals for each team when play away
uniqueIds = uniqueIdsHomeScore
uniqueIds$match_awayteam_score = uniqueIdsAwayScore$match_awayteam_score
uniqueIds$average_score = (uniqueIds$match_awayteam_score + uniqueIds$match_hometeam_score)/2 #how many goals a team scores on average for all the games it played
uniqueIds$relative_score = uniqueIds$average_score/avgGoals #the larger this value the more the goals is expected for each game the team plays


i = 1
for(i in 1:NROW(myData)){
  ind = which(uniqueIds$match_hometeam_id %in% myData$match_hometeam_id[i] )
  myData$relative_home_score[i] = uniqueIds$relative_score[ind] #assign relative scores as calculated above to the extanded data for home team
}

i = 1
for(i in 1:NROW(myData)){
  ind = which(uniqueIds$match_hometeam_id %in% myData$match_awayteam_id[i] ) #ids stored only as home team as home-away team ids are the same
  myData$relative_away_score[i] = uniqueIds$relative_score[ind] #assign relative scores as calculated above to the extanded data for home team
}

myData$relative_dif = 10*abs(myData$relative_away_score - myData$relative_home_score)

# The other variable we may consider is the scores of the last five games of each team
#for home team
i = 1
for(i in 1:NROW(myData)){
  indd = which(myData$match_hometeam_id %in% myData$match_hometeam_id[i] )
  homeTeamScore = myData$match_hometeam_score[indd]
  avgHomeTeamScore5 = mean(homeTeamScore[1:5])
  myData$avgHomeTeamScore[i] = avgHomeTeamScore5
}

#for away team
i = 1
for(i in 1:NROW(myData)){
  indd = which(myData$match_awayteam_id %in% myData$match_awayteam_id[i] )
  awayTeamScore = myData$match_awayteam_score[indd]
  avgAwayTeamScore5 = mean(awayTeamScore[1:5])
  myData$avgAwayTeamScore[i] = avgAwayTeamScore5
}
#So the factors that we consider is relative average number of goals (total average) with respect to the average considering all games for home/ away teams or the difference of them and last five number of goals for home and away teams
#I expect that if the relative score of one team is very different than the away team many goals are going to be recorded. This would have effect both on regression and classification problems
